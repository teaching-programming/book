


Sub ConvertPandocFigureTables()
    'images with titles are rendered as one-cell tables. let's untable them
    Dim i As Long
    Dim tbl As Table

    For i = ActiveDocument.Tables.Count To 1 Step -1 'backwards, for stability
        Set tbl = ActiveDocument.Tables(i)
        If tbl.Rows.Count = 1 _
           And tbl.Columns.Count = 1 _
           And tbl.Range.InlineShapes.Count > 0 Then
            tbl.ConvertToText _
                Separator:=wdSeparateByParagraphs
        End If
    Next i

End Sub


Sub ResizeSelectedImages()

    Dim para As Paragraph
    Dim tbl As Table
    Dim ils As InlineShape
    Dim shp As Shape
    Dim insertRange As Range
    Dim txt As String
    Dim NameIndex, SizeIndex As Integer
    Dim sizeColumn As Double
    Dim sizeCm As Double
    Dim prevPara As Paragraph
    NameIndex = 0
    SizeIndex = 1
    Dim targetList As Variant
    Dim targetImageData As Variant
    'here put image names which should be stretched beyond one column and by how much
    imageTransformations = Array( _
                                                Array("Figure 2.1*", 1.5), _
                                                Array("Figure 3.1*", 1.5) _
                                            )

    For Each para In ActiveDocument.Paragraphs
        txt = CleanWordText(para.Range.Text)
            If txt Like "Figure *" Then
                    targetImageData = Array("Any Image", 1)
                    For whichImage = LBound(imageTransformations) To UBound(imageTransformations)
                        If txt Like imageTransformations(whichImage)(NameIndex) Then
                              targetImageData = imageTransformations(whichImage)
                        End If
                    Next whichImage

                Set prevPara = para.Range.Previous(Unit:=wdParagraph, Count:=1).Paragraphs(1)
                'Check for image
                If prevPara.Range.InlineShapes.Count > 0 Then
                    Set ils = prevPara.Range.InlineShapes(1)
                    Set shp = ils.ConvertToShape
                    shp.LockAspectRatio = msoTrue
                    sizeColumn = 5.56
                    
                    If targetImageData(SizeIndex) <> 1 Then
                        sizeCm = sizeColumn * targetImageData(SizeIndex)
                        shp.WrapFormat.Type = wdWrapTight
                    Else
                        sizeCm = sizeColumn
                        shp.WrapFormat.Type = wdWrapInline
                    End If
                    shp.Width = CentimetersToPoints(sizeCm)
                End If
            End If

    Next para

End Sub

Function CleanWordText(txt As String) As String
        txt = Replace(txt, Chr(13), "")
        txt = Replace(txt, Chr(7), "")
        txt = Replace(txt, Chr(160), " ") 'weird space
        txt = Replace(txt, Chr(110), " ") 'weird space
        txt = Replace(txt, Chr(190), " ") 'weird space
        txt = Replace(txt, Chr(202), " ") 'weird space
        txt = Replace(txt, vbTab, " ")
        CleanWordText = txt
End Function



Sub SetListIndentation()

    Dim para As Paragraph
    Dim lf As ListFormat
    Dim lvl As ListLevel

    For Each para In ActiveDocument.Paragraphs

        Set lf = para.Range.ListFormat

        ' Check if paragraph is part of a list
        If lf.ListType <> wdListNoNumbering Then

            Set lvl = lf.ListTemplate.ListLevels(lf.ListLevelNumber)
            ' Bullet/number position
            lvl.NumberPosition = CentimetersToPoints(0)
            ' Text position
            lvl.TextPosition = CentimetersToPoints(0.3)
            ' Optional: tab follows bullet
            lvl.TabPosition = CentimetersToPoints(0.3)
        End If
    Next para
End Sub

Sub ReplaceCaptionsStyle(StyleOld As String, StyleNew As String)
    Dim para As Paragraph
    For Each para In ActiveDocument.Paragraphs
        If para.Style = StyleOld Then
            para.Style = StyleNew
        End If
    Next para
End Sub

Sub StyleNamesCleanup()
    ReplaceCaptionsStyle "Block Text", "Quote"
End Sub

Sub ShowMessage(message As String)
    Application.StatusBar = message
    DoEvents
End Sub

Sub ModifyCaptionStyle()
    Dim textStyle As Style
    Set textStyle = ActiveDocument.Styles("Image Caption")

    With textStyle.Font
        .Size = 9
        .Italic = True
        .Bold = False
        .Color = RGB(64, 64, 64)
    End With

    With textStyle.ParagraphFormat
        .SpaceBefore = 6
        .SpaceAfter = 6
    End With

End Sub

Sub ChangeFirstH2AfterH1()
    Dim para As Paragraph
    Dim foundH1 As Boolean
    Dim changed As Boolean

    foundH1 = False
    changedH2 = False

    'step over headings, and after each H1 change first H2 to heave less spacing on top
    For Each para In ActiveDocument.Paragraphs
        If para.Style = ActiveDocument.Styles("Heading 1") Then
            foundH1 = True
            changedH2 = False
        End If
        ' After first Heading 1, look for first Heading 2
        If foundH1 And Not changedH2 Then
            If para.Style = ActiveDocument.Styles("Heading 2") Then
                para.Range.Style = ActiveDocument.Styles("Heading2-Top")
                changedH2 = True
                foundH1 = False
            End If
        End If
    Next para
    ShowMessage "First Heading 2 after Heading 1 updated."

End Sub
